// TypeScript
import {{ '{' }} QueryBuilderColumn {{ '}' }} from '../../shared/components/query-builder/query-builder.component';
import {{ '{' }} RuleModel {{ '}' }} from '@syncfusion/ej2-angular-querybuilder';

columns: QueryBuilderColumn[] = [
  {{ '{' }}
    field: 'EmployeeName',
    label: 'Employee Name',
    type: 'string',
    operators: [
      {{ '{' }} key: 'equal', value: 'Equal' {{ '}' }},
      {{ '{' }} key: 'contains', value: 'Contains' {{ '}' }}
    ]
  {{ '}' }}
];

initialRule: RuleModel = {{ '{' }}
  condition: 'and',
  rules: [
    {{ '{' }}
      label: 'Employee Name',
      field: 'EmployeeName',
      type: 'string',
      operator: 'contains',
      value: 'John'
    {{ '}' }}
  ]
{{ '}' }};

// HTML
<app-query-builder
  [dataSource]="sampleData"
  [columns]="columns"
  [rule]="initialRule"
  [width]="'100%'"
  [height]="'500px'"
  (ruleChange)="onRuleChange($event)">
</app-query-builder>
