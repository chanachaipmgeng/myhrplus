// TypeScript
editorValue: string = '<p>ข้อความเริ่มต้น</p>';

toolbarSettings = {{ '{' }}
  items: [
    'Bold', 'Italic', 'Underline',
    'FontName', 'FontSize', 'FontColor',
    'Formats', 'Alignments',
    'CreateLink', 'Image'
  ]
{{ '}' }};

// HTML
<app-rich-text-editor
  [value]="editorValue"
  [toolbarSettings]="toolbarSettings"
  [height]="'400px'"
  [width]="'100%'"
  [showCharCount]="true"
  (change)="onValueChange($event)">
</app-rich-text-editor>
